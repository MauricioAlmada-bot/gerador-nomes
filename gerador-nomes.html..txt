  <script>
    // Lista de nomes (exemplo)
    const names = [
      { name: "Alice", gender: "female", origin: "greek", meaning: "Verdade, nobreza", syllables: 2 },
      { name: "Miguel", gender: "male", origin: "hebrew", meaning: "Quem é como Deus?", syllables: 2 },
      { name: "Laura", gender: "female", origin: "latin", meaning: "Loureiro, vitória", syllables: 2 },
      { name: "Davi", gender: "male", origin: "hebrew", meaning: "Amado", syllables: 2 },
      { name: "Sophia", gender: "female", origin: "greek", meaning: "Sabedoria", syllables: 3 },
      { name: "Arthur", gender: "male", origin: "celtic", meaning: "Urso nobre", syllables: 2 },
    ];

    // Função para gerar nome
    function generateName() {
      const gender = document.getElementById("gender").value;
      const origin = document.getElementById("origin").value;
      const syllables = parseInt(document.getElementById("syllables").value);

      // Filtrar nomes com base nos critérios
      let filteredNames = names.filter(item => {
        return (gender === "all" || item.gender === gender) &&
               (origin === "all" || item.origin === origin) &&
               (isNaN(syllables) || item.syllables === syllables);
      });

      // Selecionar um nome aleatório
      if (filteredNames.length > 0) {
        const randomIndex = Math.floor(Math.random() * filteredNames.length);
        const selectedName = filteredNames[randomIndex];

        // Exibir o resultado
        document.getElementById("generatedName").textContent = selectedName.name;
        document.getElementById("meaning").textContent = selectedName.meaning;
        document.getElementById("nameOrigin").textContent = selectedName.origin;
      } else {
        alert("Nenhum nome encontrado com os filtros selecionados. Tente novamente!");
      }
    }
  </script>
